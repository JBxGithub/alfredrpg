### HMM狀態引擎

```python
# 狀態定義
states = ['bull_quiet', 'bull_volatile', 'bear_quiet', 
          'bear_volatile', 'sideways_quiet', 'sideways_volatile']

# 轉移概率矩陣（典型值）
transition_matrix = [
    [0.80, 0.10, 0.05, 0.03, 0.02, 0.00],  # bull_quiet
    [0.15, 0.75, 0.03, 0.05, 0.02, 0.00],  # bull_volatile
    [0.05, 0.03, 0.80, 0.10, 0.02, 0.00],  # bear_quiet
    [0.03, 0.05, 0.15, 0.75, 0.02, 0.00],  # bear_volatile
    [0.02, 0.02, 0.02, 0.02, 0.85, 0.07],  # sideways_quiet
    [0.02, 0.02, 0.02, 0.02, 0.10, 0.82],  # sideways_volatile
]

# 觀測特徵
features = ['returns', 'volatility', 'volume', 'vix']
```

### 經濟週期定位器

```python
def detect_cycle_phase(pmi, yield_curve, unemployment):
    """
    基於經典經濟指標定位週期階段
    """
    if pmi > 50 and yield_curve > 0 and unemployment < 5:
        return 'early_cycle'
    elif pmi > 50 and yield_curve < 0.5:
        return 'mid_cycle'
    elif pmi < 50 and yield_curve < 0:
        return 'late_cycle'
    else:
        return 'recession'
```

### 板塊輪動追蹤器

```python
def sector_momentum_ranking(sector_returns, lookback=252):
    """
    12個月動量排名（skip最近1個月避免反轉）
    """
    momentum_12m = sector_returns.iloc[-252:-21].mean()
    momentum_1m = sector_returns.iloc[-21:].mean()
    
    # 標準動量（排除最近1個月）
    standard_momentum = momentum_12m - momentum_1m
    
    return standard_momentum.rank(ascending=False)
```

### 動態止損計算

```python
def dynamic_stop_loss(entry_price, atr, market_state):
    """
    基於狀態的動態止損
    """
    multipliers = {
        'bull_quiet': 2.0,
        'bull_volatile': 3.0,
        'sideways_quiet': 2.5,
        'sideways_volatile': 3.5,
        'bear_quiet': 2.0,
        'bear_volatile': 4.0
    }
    
    return entry_price - (atr * multipliers[market_state])
```
